3. SELECT podzespoly.nazwa, producenci.nazwa FROM podzespoly, producenci WHERE podzespoly.producenci_id = producenci.id;
4. UPDATE producenci SET nazwa = 'PATRIOT' WHERE nazwa = 'Patriot';
